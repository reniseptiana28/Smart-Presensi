Script Google Sheets (FIX TANGGAL)

Salin kode di bawah ini ke Apps Script. Kode ini akan menangkap tanggal dan waktu dari aplikasi secara akurat:

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["TANGGAL", "WAKTU", "NAMA PEGAWAI", "UNIT KERJA", "STATUS"]);
    sheet.getRange("A1:E1").setFontWeight("bold").setBackground("#064e3b").setFontColor("#ffffff");
  }
  
  // Menggunakan data tanggal dan waktu yang dikirim dari aplikasi
  // Atau menggunakan waktu server Google jika data tidak ada
  var tgl = data.tanggal || new Date().toLocaleDateString('id-ID');
  var wkt = data.waktu || new Date().toLocaleTimeString('id-ID');

  sheet.appendRow([tgl, wkt, data.nama, data.unit, data.status]);
  
  return ContentService.createTextOutput("Success");
}

Penting: Pastikan Anda klik "Deploy" -> "New Deployment" -> "Web App" dan pilih "Anyone" pada "Who has access".
